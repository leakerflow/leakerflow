// Analyze leak credibility
function analyzeLeak(leakText) {
  try {
    // Determine source type
    const sourceType = determineSourceType(leakText);
    const sourceData = CREDIBILITY_SOURCES[sourceType];
    
    // Calculate technical feasibility
    const techFeasibility = calculateTechnicalFeasibility(leakText);
    
    // Calculate community consensus
    const communityConsensus = calculateCommunityConsensus(leakText);
    
    // Calculate overall credibility
    const overallScore = Math.round((sourceData.score + techFeasibility + communityConsensus) / 3);
    
    // Display results
    displayAnalysisResults({
      leakText,
      sourceType: sourceData.type,
      sourceScore: sourceData.score,
      techScore: techFeasibility,
      communityScore: communityConsensus,
      overallScore
    });
    
    // Clear input
    leakInput.value = '';
    
  } catch (error) {
    console.error('Error analyzing leak:', error);
    alert('Could not analyze leak. Please try again.');
  }
}

// Determine source type based on leak text
function determineSourceType(text) {
  const lowerText = text.toLowerCase();
  
  if (lowerText.includes('rockstar') || lowerText.includes('official')) {
    return 'rockstar_official';
  } else if (lowerText.includes('tom henderson') || lowerText.includes('_tom_henderson_')) {
    return 'tom_henderson';
  } else if (lowerText.includes('jason schreier') || lowerText.includes('schreier')) {
    return 'jason_schreier';
  } else if (lowerText.includes('gtaforums') || lowerText.includes('verified insider')) {
    return 'gtaforums_verified';
  } else if (lowerText.includes('reddit') || lowerText.includes('r/gta')) {
    return 'reddit_leak';
  } else if (lowerText.includes('4chan') || lowerText.includes('anonymous')) {
    return 'anonymous_4chan';
  } else {
    return 'social_media';
  }
}

// Calculate technical feasibility score
function calculateTechnicalFeasibility(text) {
  const lowerText = text.toLowerCase();
  let score = 50; // Base score
  
  // Technical indicators that increase credibility
  if (lowerText.includes('engine') || lowerText.includes('rage')) score += 15;
  if (lowerText.includes('console') || lowerText.includes('ps5') || lowerText.includes('xbox')) score += 10;
  if (lowerText.includes('development') || lowerText.includes('build')) score += 10;
  if (lowerText.includes('screenshot') || lowerText.includes('footage')) score += 20;
  
  // Unrealistic claims that decrease credibility
  if (lowerText.includes('entire map') || lowerText.includes('all cities')) score -= 20;
  if (lowerText.includes('release tomorrow') || lowerText.includes('next week')) score -= 30;
  if (lowerText.includes('100gb') || lowerText.includes('1000 hours')) score -= 15;
  
  return Math.max(0, Math.min(100, score));
}

// Calculate community consensus score
function calculateCommunityConsensus(text) {
  const lowerText = text.toLowerCase();
  let score = 50; // Base score
  
  // Community validation indicators
  if (lowerText.includes('corroborated') || lowerText.includes('confirmed')) score += 20;
  if (lowerText.includes('multiple sources') || lowerText.includes('verified')) score += 15;
  if (lowerText.includes('community') || lowerText.includes('discussion')) score += 10;
  
  // Skepticism indicators
  if (lowerText.includes('debunked') || lowerText.includes('fake')) score -= 40;
  if (lowerText.includes('speculation') || lowerText.includes('rumor')) score -= 10;
  if (lowerText.includes('unconfirmed') || lowerText.includes('alleged')) score -= 15;
  
  return Math.max(0, Math.min(100, score));
}

// Display analysis results
function displayAnalysisResults(results) {
  leakTitle.textContent = 'Leak Analysis Results';
  credibilityLevel.textContent = getCredibilityLabel(results.overallScore);
  assessmentDescription.textContent = getAssessmentDescription(results);
  
  // Update scores
  sourceScore.textContent = `${results.sourceScore}/100`;
  techScore.textContent = `${results.techScore}/100`;
  communityScore.textContent = `${results.communityScore}/100`;
  
  // Display credibility score with color coding
  credibilityScore.innerHTML = `<div class="score-circle ${getScoreClass(results.overallScore)}">${results.overallScore}</div>`;
}

// Get credibility label based on score
function getCredibilityLabel(score) {
  if (score >= 80) return 'HIGHLY CREDIBLE';
  if (score >= 60) return 'MODERATELY CREDIBLE';
  if (score >= 40) return 'LOW CREDIBILITY';
  return 'HIGHLY QUESTIONABLE';
}

// Get score class for styling
function getScoreClass(score) {
  if (score >= 80) return 'high-credibility';
  if (score >= 60) return 'medium-credibility';
  if (score >= 40) return 'low-credibility';
  return 'very-low-credibility';
}

// Get detailed assessment description
function getAssessmentDescription(results) {
  let description = `Source Type: ${results.sourceType}. `;
  
  if (results.overallScore >= 80) {
    description += 'This leak shows strong indicators of authenticity and should be taken seriously.';
  } else if (results.overallScore >= 60) {
    description += 'This leak has moderate credibility but requires additional verification.';
  } else if (results.overallScore >= 40) {
    description += 'This leak has low credibility and should be treated with significant skepticism.';
  } else {
    description += 'This leak appears highly questionable and likely unreliable.';
  }
  
  return description;
 }
